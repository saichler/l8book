Практика: Делаем архитектуру явной

На этом этапе у вас есть набор сообщений Protocol Buffer, и вы уже определили Первичные объекты в вашей модели.

Используя пример из предыдущей главы, следующий фрагмент показывает упрощённое представление сгенерированных определений protobuf:

message EmployeeList {
  repeated Employee list = 1;
  l8api.L8MetaData metadata = 2;
}

// Первичный объект
message Employee {
  string id = 1;
  string name = 2;
  repeated Addr addresses = 3;
}

message AddressList {
  repeated Employee list = 1;
  l8api.L8MetaData metadata = 2;
}

// Первичный объект
message Addr {
  string line1 = 1;
  string line2 = 2;
  int32 zip = 3;
}

На этом этапе важно не синтаксис сообщений, а явная идентификация Первичных объектов. Первичные объекты определяют владение жизненным циклом, границы конкурентности и ответственность сервиса в Layer 8.

Примечание: Layer 8 включает скрипт, упрощающий генерацию привязок Protocol Buffer для рабочих процессов с ИИ-ассистентом. Этот скрипт будет рассмотрен в следующей главе.

Определение и активация SLA

В этом примере мы используем распределённый кеш-сервис. Использование персистентного сервиса следует тому же паттерну, с единственным отличием — инъекцией реализации слоя персистентности.

Ниже приведён метод активации для сервиса Employee. Здесь архитектура становится явной.

package employees

import (
    "github.com/saichler/l8erp/go/types/hcm"
    "github.com/saichler/l8services/go/services/base"
    "github.com/saichler/l8types/go/ifs"
    "github.com/saichler/l8types/go/types/l8api"
    "github.com/saichler/l8types/go/types/l8web"
    "github.com/saichler/l8utils/go/utils/web"
)

func ActivateEmployees(vnic ifs.IVNic) {
    // Логическое имя сервиса, используемое другими сервисами для ссылки на этот сервис
    serviceName := "Employee"

    // Область сервиса обеспечивает вертикальное масштабирование (например, несколько изолированных кешей в одном процессе)
    serviceArea := byte(0)

    sla := ifs.NewServiceLevelAgreement(
        &base.BaseService{},      // Базовая реализация сервиса Layer 8
        serviceName, serviceArea, // Логическая идентичность сервиса
        true,                     // Сервис с состоянием (кеш в памяти)
        nil,                      // Опциональный callback для валидации или предобработки
    )

    // Определяем Первичный объект и его коллекцию
    sla.SetServiceItem(&hcm.Employee{})
    sla.SetServiceItemList(&hcm.EmployeeList{})

    // Определяем первичный ключ, уникально идентифицирующий Первичный объект
    sla.SetPrimaryKeys("EmployeeId")

    // Выбираем модель конкурентности
    sla.SetTransactional(true) // Транзакционная vs best-effort конкурентность

    // Отключаем репликацию; все экземпляры поддерживают идентичное состояние
    sla.SetReplication(false)

    // Определяем поверхность Web API для этого сервиса
    ws := web.New(serviceName, serviceArea, 0)

    // CRUD-стиль эндпоинты, отображённые на семантику Первичного объекта
    ws.AddEndpoint(&hcm.Employee{}, ifs.POST, &l8web.L8Empty{})
    ws.AddEndpoint(&hcm.EmployeeList{}, ifs.POST, &l8web.L8Empty{})
    ws.AddEndpoint(&hcm.Employee{}, ifs.PUT, &l8web.L8Empty{})
    ws.AddEndpoint(&hcm.Employee{}, ifs.PATCH, &l8web.L8Empty{})
    ws.AddEndpoint(&l8api.L8Query{}, ifs.DELETE, &l8web.L8Empty{})
    ws.AddEndpoint(&l8api.L8Query{}, ifs.GET, &hcm.EmployeeList{})

    sla.SetWebService(ws)

    // Активируем сервис в контексте виртуальной сети
    vnic.Resources().Services().Activate(sla, vnic)
}

Здесь важна не механика API, а архитектурная декларация:

  • Первичный объект явен
  • Модель конкурентности явна
  • Жизненный цикл сервиса явен
  • Поверхность API выводится из модели, а не из ручной логики

На этом этапе система больше не полагается на неявные соглашения. Архитектура декларирована, обеспечена и наблюдаема.

Создание ресурсов

Resources инкапсулируют экосистемные сервисы, требуемые системой на базе Layer 8. Они формируют фундамент времени выполнения, на котором строятся сетевое взаимодействие, безопасность, управление сервисами и наблюдаемость.

В большинстве проектов создание ресурсов централизовано в одном статическом или bootstrap-методе, аналогично примеру ниже. Этот метод обычно вызывается один раз при инициализации процесса.

// CreateResources инициализирует и связывает все основные экосистемные сервисы,
// требуемые этим проектом.
func CreateResources(alias string) ifs.IResources {
    // Создаём и конфигурируем логгер
    log := logger.NewLoggerImpl(&logger.FmtLogMethod{})
    log.SetLogLevel(ifs.Info_Level)

    // Создаём контейнер Resources
    res := resources.NewResources(log)

    // Регистрируем реестр сервисов
    res.Set(registry.NewRegistry())

    // Загружаем и инициализируем провайдера безопасности
    sec, err := ifs.LoadSecurityProvider(res)
    if err != nil {
        // Задержка для сброса логов перед крахом
        time.Sleep(time.Second * 10)
        panic(err.Error())
    }
    res.Set(sec)

    // Конфигурируем сетевое взаимодействие с использованием значений по умолчанию
    conf := &l8sysconfig.L8SysConfig{
        MaxDataSize:               resources.DEFAULT_MAX_DATA_SIZE,
        RxQueueSize:               resources.DEFAULT_QUEUE_SIZE,
        TxQueueSize:               resources.DEFAULT_QUEUE_SIZE,
        LocalAlias:                alias,        // Идентификатор пода или процесса для наблюдаемости
        VnetPort:                  uint32(20201), // Порт виртуальной сети
        KeepAliveIntervalSeconds:  30,            // Интервал heartbeat к vNet
    }
    res.Set(conf)

    // Включаем интроспекцию времени выполнения
    res.Set(introspecting.NewIntrospect(res.Registry()))

    // Инициализируем менеджер сервисов
    res.Set(manager.NewServices(res))

    return res
}

Содержащий процесс

Содержащий процесс отвечает за создание виртуального сетевого интерфейса (vNic) и активацию сервисов в этом контексте времени выполнения.

С точки зрения приложения этот процесс следует маленькой и детерминированной последовательности: инициализация ресурсов, установление сетевого взаимодействия, активация сервисов и ожидание.

func main() {
    // Создаём и инициализируем все экосистемные ресурсы
    res := CreateResources("pod name / process name / container name")

    // Выбираем режим сетевого взаимодействия.
    // При работе под Kubernetes это включает нативную интеграцию.
    // Если Kubernetes отсутствует, фреймворк автоматически
    // переключается на сетевое взаимодействие уровня контейнера, затем процесса.
    ifs.SetNetworkMode(ifs.NETWORK_K8s)

    // Создаём виртуальный сетевой интерфейс (vNic)
    nic := vnic.NewVirtualNetworkInterface(res, nil)

    // Запускаем vNic и его базовый сетевой стек
    nic.Start()

    // Блокируемся до подключения и готовности vNic
    nic.WaitForConnection()

    // Активируем сервисы в этом сетевом контексте
    employees.ActivateEmployees(nic)

    // Блокируемся до получения сигнала завершения
    // (например, SIGTERM, SIGINT). Очистка обрабатывается фреймворком.
    common.WaitForSignal(res)
}

Запуск веб-сервера

Запуск веб-сервера для предоставления Web API, включая эндпоинты аутентификации, намеренно так же прост, как активация любого другого сервиса.

Он следует точно такому же архитектурному паттерну: определяется SLA, активируется и привязывается к контексту виртуальной сети с небольшим числом специфичных для веба шагов конфигурации.

Следующий пример можно использовать вместо вызова employees.ActivateEmployees(nic), когда процесс отвечает за обслуживание HTTP-эндпоинтов.

func startWebServer(nic ifs.IVNic, httpPort int, cert string) {
    // Конфигурируем REST-сервер
    serverConfig := &server.RestServerConfig{
        Host:           ipsegment.MachineIP, // Определяем IP машины и привязываемся к нему
        Port:           httpPort,            // HTTP/HTTPS порт прослушивания
        Authentication: true,                // Включаем безопасную коммуникацию
        CertName:       cert,                // Идентификатор TLS-сертификата
        Prefix:         "/myApp/",            // Префикс URL приложения
    }

    // Создаём экземпляр REST-сервера
    svr, err := server.NewRestServer(serverConfig)
    if err != nil {
        panic(err)
    }

    // Регистрируем встроенные эндпоинты health-сервиса
    hs, ok := nic.Resources().Services().ServiceHandler(health.ServiceName, 0)
    if ok {
        ws := hs.WebService()
        svr.RegisterWebService(ws, nic)
    }

    // Активируем веб-сервис как первоклассный сервис Layer 8
    sla := ifs.NewServiceLevelAgreement(
        &server.WebService{},
        ifs.WebService,
        0,
        false,
        nil,
    )
    sla.SetArgs(svr)

    nic.Resources().Services().Activate(sla, nic)

    nic.Resources().Logger().Info("Web server started")

    // Начинаем принимать HTTP-запросы
    svr.Start()
}
